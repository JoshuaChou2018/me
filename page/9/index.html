class Solution(object):
    def minMeetingRooms(self, intervals):
        """
        :type intervals: List[List[int]]
        :rtype: int
        """
        dic={}
        for interval in intervals:
            try:
                dic[interval[0]]+=1
            except:
                dic[interval[0]]=1
            try:
                dic[interval[1]]-=1
            except:
                dic[interval[1]]=-1
        max_=0
        count=0
        for key in sorted(dic.keys()):
            count+=dic[key]
            if count>max_:
                max_=count
        return max_
